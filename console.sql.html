48   --za prikaz vseh drzav:
49   CREATE VIEW view_drzave AS
50   SELECT * FROM drzave;
51   
52   --za prikaz vseh zaloznikov:
53   CREATE VIEW view_zalozniki AS
54   SELECT z.zalozniki_id, z.drzave_id, d.drzava_lokalno, z.zaloznik
55   FROM zalozniki z
56   INNER JOIN drzave d ON z.drzave_id = d.drzave_id;
57   
58   
59   -- za prikaz vseh ključnih besed:
60   CREATE VIEW view_kljucne_besede AS
61   SELECT * FROM kljucne_besede;
62   
63   --za prikaz vseh jezikov:
64   CREATE VIEW view_jeziki AS
65   SELECT * FROM jeziki;
66   
67   
68   --za prikaz vseh knjig:
69   CREATE VIEW view_knjige_gradivo AS
70   SELECT k.knjige_gradivo_id, z.zaloznik, j.jezik, k.naslov, k.leto_izdaje, k.opis, k.fizicni_opis, k.opombe, k.isbn
71   FROM knjige_gradivo k
72   INNER JOIN zalozniki z ON k.zalozniki_id = z.zalozniki_id
73   INNER JOIN jeziki j ON k.jeziki_id = j.jeziki_id;
74   
75   --za prikaz vseh oseb:
76   CREATE VIEW view_osebe AS
77   SELECT * FROM osebe;
78   
79   -- za prikaz vseh uporabniških računov:
80   CREATE VIEW view_uporabniski_racuni AS
81   SELECT u.uporabniski_racuni_id, o.ime, o.priimek, u.uporabnisko_ime, u.vrsta_racuna, u.aktiven, u.e_naslov
82   FROM uporabniski_racuni u
83   INNER JOIN osebe o ON u.osebe_id = o.osebe_id;
84   
85   --za prikaz ocen knjig:
86   CREATE VIEW view_knjige_gradivo_ocene AS
87   SELECT k.naslov, o.ime, o.priimek, ose.uporabnisko_ime, kgo.ocena
88   FROM knjige_gradivo_ocene kgo
89   INNER JOIN knjige_gradivo k ON kgo.knjige_gradivo_id = k.knjige_gradivo_id
90   INNER JOIN uporabniski_racuni ose ON kgo.uporabniski_racuni_id = ose.uporabniski_racuni_id
91   INNER JOIN osebe o ON ose.osebe_id = o.osebe_id;
